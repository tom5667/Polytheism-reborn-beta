	color2 = { 255 255 255 }
	definite_form = yes 
	ruler_uses_title_name = yes
	landless = no
	capital = c_dublin

	can_create = {
		trigger_if = {
			limit = {
				is_ai = yes
			}
			culture = { has_cultural_pillar = heritage_goidelic }
			faith = faith:gaelic_polytheism
		}
	}
	
	ai_primary_priority = {
		if = {

		if = {

			}

		}
	}	
}
