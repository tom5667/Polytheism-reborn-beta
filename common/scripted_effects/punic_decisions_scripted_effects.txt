return_punic_sr_decision_scripted_effect = {
	#Change yourself, and any willing vassals/family, over to Gaelic polytheism.

	set_character_faith_with_conversion = faith:punic_polytheism
	
	#Gain nickname for your troubles.
	give_nickname = nick_the_apostate
	#give trait restored Gaelic polytheism as state religion trait
	add_trait = punic_state_religion
	add_trait = zealous
	remove_trait = pilgrim
	clear_global_variable_list = unavailable_unique_decisions_convert
	#berbergroup counties of your old religion may defect back to the old ways.
	custom_tooltip = return_punic_sr_decision_scripted_effect.county_conversions.tt
	hidden_effect = {
		primary_title = {
			every_in_de_facto_hierarchy = {
				limit = {
					tier = tier_county
					culture_group = culture_group:berber_group
					religion = scope:old_faith.religion
				}
				#berber remember the old ways best, and have a high chance to flip.
				if = {
					limit = { culture = culture:baranis }
					random = {
						chance = 45
						set_county_faith = faith:punic_polytheism
					}
				}
				#Slightly up the chances for the apostate.
				else_if = {
					limit = { holder = root }
					random = {
						chance = 35
						set_county_faith = faith:punic_polytheism
					}
				}
				#Other berber cultures still have a moderate chance to flip.
				else = {
					random = {
						chance = 15
						set_county_faith = faith:punic_polytheism
					}
				}
			}
		}
	}
	#Make flipping counties deliriously happy about the whole affair.
	primary_title = {
		every_in_de_facto_hierarchy = {
			limit = {
				tier = tier_county
				faith = faith:punic_polytheism
			}
			custom = punic_resettlement.every_converted_province
			add_county_modifier = {
				modifier = punic_appreciation_modifier
				years = 25
			}
		}
	}
	#Finally, give punic_polytheist a hefty fervour boost, just so that they don't immediately convert back.
	# hidden_effect = {
	# 	faith:punic_polytheism = { change_fervor = 100 }
	# }
}
